```php
<?php

namespace Tests\Unit\Contexts\Geography\ValueObjects;

use App\Contexts\Geography\Domain\ValueObjects\CountryCode;
use App\Contexts\Geography\Domain\ValueObjects\GeographyHierarchy;
use InvalidArgumentException;
use PHPUnit\Framework\TestCase;

class GeographyHierarchyTest extends TestCase
{
    /** @test */
    public function it_creates_valid_hierarchy(): void
    {
        // Arrange
        $countryCode = CountryCode::fromString('NP');
        $levelIds = [1, 12, 123, null, null, null, null, null];

        // Act
        $hierarchy = GeographyHierarchy::fromLevelIds($countryCode, $levelIds);

        // Assert
        $this->assertInstanceOf(GeographyHierarchy::class, $hierarchy);
        $this->assertEquals('NP', $hierarchy->getCountryCode()->toString());
        $this->assertEquals([1, 12, 123, null, null, null, null, null], $hierarchy->getLevelIds());
    }

    /** @test */
    public function it_throws_exception_for_hierarchy_with_gap(): void
    {
        // Arrange
        $countryCode = CountryCode::fromString('NP');

        // Assert
        $this->expectException(InvalidArgumentException::class);
        $this->expectExceptionMessage('Missing level 2 in hierarchy. Cannot have level 3 without level 2.');

        // Act - Level 1 filled, level 2 missing (gap), level 3 filled
        GeographyHierarchy::fromLevelIds($countryCode, [1, null, 3]);
    }

    /** @test */
    public function it_throws_exception_for_missing_required_levels(): void
    {
        // Arrange
        $countryCode = CountryCode::fromString('NP');

        // Assert - Nepal requires levels 1-2
        $this->expectException(InvalidArgumentException::class);
        $this->expectExceptionMessage('Level 1 (Province) is required for country NP.');

        // Act - Missing required level 1
        GeographyHierarchy::fromLevelIds($countryCode, [null, 12]);
    }

    /** @test */
    public function it_converts_to_geo_path(): void
    {
        // Arrange
        $countryCode = CountryCode::fromString('NP');
        $hierarchy = GeographyHierarchy::fromLevelIds($countryCode, [1, 12, 123]);

        // Act
        $geoPath = $hierarchy->toGeoPath();

        // Assert
        $this->assertEquals('1.12.123', $geoPath->toString());
    }

    /** @test */
    public function it_gets_filled_levels(): void
    {
        // Arrange
        $countryCode = CountryCode::fromString('NP');
        $hierarchy = GeographyHierarchy::fromLevelIds($countryCode, [1, 12, null, 1234]);

        // Act
        $filledLevels = $hierarchy->getFilledLevels();

        // Assert
        $this->assertEquals([1, 2, 4], $filledLevels);
    }

    /** @test */
    public function it_gets_missing_levels(): void
    {
        // Arrange
        $countryCode = CountryCode::fromString('NP');
        $hierarchy = GeographyHierarchy::fromLevelIds($countryCode, [1, null, 123, null]);

        // Act
        $missingLevels = $hierarchy->getMissingLevels();

        // Assert
        $this->assertEquals([2, 4, 5, 6, 7, 8], $missingLevels);
    }

    /** @test */
    public function it_checks_if_complete(): void
    {
        // Arrange
        $countryCode = CountryCode::fromString('NP');
        
        $incomplete = GeographyHierarchy::fromLevelIds($countryCode, [1, 12, null]);
        $complete = GeographyHierarchy::fromLevelIds($countryCode, [1, 12, 123, 1234, 12345, 123456, 1234567, 12345678]);

        // Assert
        $this->assertFalse($incomplete->isComplete());
        $this->assertTrue($complete->isComplete());
    }

    /** @test */
    public function it_creates_empty_hierarchy(): void
    {
        // Arrange
        $countryCode = CountryCode::fromString('NP');

        // Act
        $hierarchy = GeographyHierarchy::empty($countryCode);

        // Assert
        $this->assertTrue($hierarchy->isEmpty());
        $this->assertEquals([null, null, null, null, null, null, null, null], $hierarchy->getLevelIds());
    }
}
```
